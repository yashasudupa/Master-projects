75 void ExeGraphBBTime<G>::processWorkSpace(WorkSpace *fw) {
76  bool built = false, reset = false;
77  this->fw = fw;
78 
79  // If required, find the microprocessor
80  if(!microprocessor) {
81  microprocessor = PROCESSOR(fw);
82  if(microprocessor)
83  reset = true;
84  else {
85  const hard::Processor *proc = hard::PROCESSOR(fw);
86  if(!proc)
87  throw ProcessorException(*this, "no processor to work with");
88  else {
89  reset = true;
90  built = true;
91  microprocessor = new Microprocessor<ExeNode>(proc);
92  }
93  }
94  }
95  //elm::cerr << "=> microprocessor = " << microprocessor << io::endl;
96 
97  // Perform the actual process
98  BBProcessor::processWorkSpace(fw);
99 
100  // Cleanup if required
101  if(built)
102  delete microprocessor;
103  if(reset)
104  microprocessor = 0;
105 }
