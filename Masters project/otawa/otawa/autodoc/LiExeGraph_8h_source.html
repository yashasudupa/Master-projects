73 inline LiExeNode::LiExeNode(ExeGraph<LiExeNode> *graph, PipelineStage<LiExeNode> *stage, ExeInst<LiExeNode> *inst)
74  : ExeNode(graph, stage, inst), _shaded(false), _graph((LiExeGraph *)graph) {
75  if (inst->codePart() >= ExeGraph<LiExeNode>::BODY) {
76  setTime(ExeGraph<LiExeNode>::READY,ExeGraph<LiExeNode>::MIN,0);
77  setTime(ExeGraph<LiExeNode>::START,ExeGraph<LiExeNode>::MIN,0);
78  setTime(ExeGraph<LiExeNode>::FINISH,ExeGraph<LiExeNode>::MIN,stage->minLatency());
79  }
80  else {
81  setTime(ExeGraph<LiExeNode>::READY,ExeGraph<LiExeNode>::MIN,-INFINITE_TIME);
82  setTime(ExeGraph<LiExeNode>::START,ExeGraph<LiExeNode>::MIN,-INFINITE_TIME);
