37  void start(BasicBlock *bb) {
38  i.start(bb);
39  man.free(s);
40  man.free(es);
41  s = man.copy(man.state(bb));
42  es = man.bot();
43  for(int i = 0; i < ss.count(); i++)
44  man.free(ss[i]);
45  ss.clear();
46  }
47 
48  void next(void) {
49  if(i.ended())
50  return;
51 
52  if(i.pathEnd()) { // semantic path end
53  es = man.joinPath(es, s); // join path state
54  if(ss) // end of semantic path
55  s = ss.pop();
56  else { // end of instruction
57  s = es;
58  es = man.bot();
59  }
60  }
61 
62  else { // next semantic instruction
63  if(i.isCond())
64  ss.push(man.copy(s));
65  s = man.updatePath(s, *i);
66  }
67 
68  i.next();
69  }
70 
71  inline void nextInst(void) { next(); while(!i.instEnd()) next(); }
72  inline void toEnd(void) { while(!i.ended()) next(); }
