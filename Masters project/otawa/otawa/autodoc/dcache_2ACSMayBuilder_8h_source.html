60  ASSERT((id >= 0) && (id < size));
61  if (age[id] == -1)
62  return;

64  if (age[id] >= A)
65  age[id] = -1;

68  inline void inject(const int id) {
69  if (contains(id)) {
70  for (int i = 0; i < size; i++)
71  if ((age[i] <= age[id]) && (age[i] != -1))
72  age[i]++;
73  }
74  else
75  for (int i = 0; i < size; i++) {
76  if (age[i] != -1)
